/*INSTALAR NGINX*/
sudo apt update
sudo apt install nginx

system status nginx
//devuelve Active: active (running)
//se puede ingresar a nginx por medio de la ip publica

/*CONFIGURAR COMO LOADBALANCER*/

sudo touch /etc/nginx/conf.d/load-balancer.conf
sudo rm -r /etc/nginx/sites-enabled/default
sudo nano /etc/nginx/conf.d/load-balancer.conf
//si hay problema al guardar el archivo
	sudo chmod 777 /etc/nginx/conf.d
	sudo chmod 777 /etc/nginx/conf.d/load-balancer.conf
/*INI CONTENIDO DEL ARCHIVO*/
upstream backend {
	server IP:PUERTO;
	server IP:PUERTO;
}

server {
	listen 80;
	location / {
		proxy_pass http://backend;
	}
}
/*FIN CONTENIDO DEL ARCHIVO*/

sudo systemctl restart nginx
//se puede ingresar al balanceador por medio de la ip
//metodo por default round robin
